> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Replace `/policy` with `/cockpit` — a browser-based flywheel control center with a fixed shell (zero-token pipeline/NOTES/scenario operations) and CC-generated dynamic component slots.

**Architecture:** `repl_admin.py serve` starts a local HTTP server; the fixed `cockpit_shell.html` polls `/api/policy` every 5s and manages an action queue; submitting the queue writes `pending-actions.json` atomically; `PendingActionMonitor` in `emerge_daemon.py` detects the file and notifies CC via `notifications/claude/channel`; CC executes actions via subagents.

**Tech Stack:** Python stdlib (`http.server`, `threading`, `socketserver`), vanilla JS (no framework), existing `repl_admin.py` and `emerge_daemon.py` patterns.

```python
def _format_pending_actions_message(actions: list) -> str:
    lines = ["[Cockpit] 用户提交了以下操作，请依次执行："]
    for i, a in enumerate(actions, 1):
        t = a.get("type", "unknown")
        if t == "pipeline-set":
            lines.append(f"{i}. pipeline-set {a.get('key')} fields={a.get('fields', {})}")
        elif t == "pipeline-delete":
            lines.append(f"{i}. pipeline-delete {a.get('key')}")
        elif t == "notes-edit":
            lines.append(f"{i}. 更新 {a.get('connector')} NOTES.md（全文替换）")
        elif t == "notes-comment":
            lines.append(f"{i}. 追加 comment 到 {a.get('connector')} NOTES.md: {str(a.get('comment', ''))[:80]}")
        elif t == "scenario-run":
            lines.append(f"{i}. 运行 scenario {a.get('scenario')} (connector: {a.get('connector')}) args={a.get('args', {})}")
        elif t == "crystallize-component":
            lines.append(f"{i}. 固化组件 {a.get('filename')} → {a.get('connector')}/cockpit/")
        else:
            lines.append(f"{i}. {t}: {a}")
    return "\n".join(lines)


class PendingActionMonitor(threading.Thread):
    """Polls ~/.emerge/repl/pending-actions.json every 2s.
    When a new submission is detected, fires a notifications/claude/channel
    push so CC processes the pending actions via subagents.
    """

    def __init__(self, state_root: Path, write_push_fn) -> None:
        super().__init__(daemon=True, name="PendingActionMonitor")
        self._state_root = state_root
        self._write_push_fn = write_push_fn
        self._stop_event = threading.Event()
        self._last_seen_ts: int = 0

    def run(self) -> None:
        pending_path = self._state_root / "pending-actions.json"
        while not self._stop_event.wait(2.0):
            if not pending_path.exists():
                continue
            try:
                data = json.loads(pending_path.read_text(encoding="utf-8"))
                ts = int(data.get("submitted_at", 0))
                if ts <= self._last_seen_ts:
                    continue
                self._last_seen_ts = ts
                actions = data.get("actions", [])
                self._write_push_fn({
                    "jsonrpc": "2.0",
                    "method": "notifications/claude/channel",
                    "params": {
                        "serverName": "emerge",
                        "content": _format_pending_actions_message(actions),
                        "meta": {
                            "source": "cockpit",
                            "action_count": len(actions),
                            "action_types": list({a.get("type") for a in actions}),
                        },
                    },
                })
                processed = self._state_root / "pending-actions.processed.json"
                pending_path.rename(processed)
            except Exception:
                pass

    def stop(self) -> None:
        self._stop_event.set()
```

```python
def _extract_scenario_args(scenario_data: dict) -> list[str]:
    """Return required arg names by scanning {{ token }} in scenario YAML."""
    import re
    text = json.dumps(scenario_data)
    tokens = set(re.findall(r"\{\{\s*([\w]+)\s*(?:\|[^}]*)?\}\}", text))
    derived: set[str] = set()
    for step in scenario_data.get("steps", []):
        if step.get("type") == "derive":
            derived.update(step.get("compute", {}).keys())
    # Remove derived, skip_if_arg values, and boolean literals
    skip_args: set[str] = set()
    for step in scenario_data.get("steps", []):
        for k in ("skip_if_arg", "skip_if_arg_missing"):
            if step.get(k):
                skip_args.add(step[k])
    return sorted(tokens - derived - skip_args - {"true", "false", "null"})


def cmd_assets() -> dict:
    """Return per-connector assets: notes content, scenario metadata, crystallized components."""
    try:
        connector_root = _resolve_connector_root()
    except Exception:
        return {"connectors": {}}

    connectors: dict = {}
    if not connector_root.exists():
        return {"connectors": connectors}

    for connector_dir in sorted(connector_root.iterdir()):
        if not connector_dir.is_dir():
            continue
        name = connector_dir.name

        notes_path = connector_dir / "NOTES.md"
        notes = notes_path.read_text(encoding="utf-8") if notes_path.exists() else None

        scenarios: list = []
        scenarios_dir = connector_dir / "scenarios"
        if scenarios_dir.exists():
            for f in sorted(scenarios_dir.iterdir()):
                if f.suffix not in (".yaml", ".yml"):
                    continue
                try:
                    import yaml as _yaml  # type: ignore[import]
                    data = _yaml.safe_load(f.read_text(encoding="utf-8"))
                except ImportError:
                    data = None
                except Exception:
                    data = None
                if not isinstance(data, dict):
                    continue
                scenarios.append({
                    "name": data.get("name", f.stem),
                    "description": (data.get("description") or "").strip(),
                    "filename": f.name,
                    "step_count": len(data.get("steps", [])),
                    "has_rollback": bool(data.get("rollback")),
                    "required_args": _extract_scenario_args(data),
                })

        components: list = []
        cockpit_dir = connector_dir / "cockpit"
        if cockpit_dir.exists():
            for html_file in sorted(cockpit_dir.glob("*.html")):
                ctx_file = cockpit_dir / f"{html_file.stem}.context.md"
                components.append({
                    "filename": html_file.name,
                    "context": ctx_file.read_text(encoding="utf-8") if ctx_file.exists() else "",
                })

        connectors[name] = {"notes": notes, "scenarios": scenarios, "components": components}

    return {"connectors": connectors}


def cmd_submit_actions(actions: list) -> dict:
    """Atomically write pending-actions.json to trigger PendingActionMonitor."""
    state_root = _resolve_state_root()
    state_root.mkdir(parents=True, exist_ok=True)
    pending_path = state_root / "pending-actions.json"
    tmp_path = state_root / "pending-actions.json.tmp"
    payload = {
        "submitted_at": int(time.time() * 1000),
        "actions": actions,
    }
    tmp_path.write_text(json.dumps(payload, indent=2, ensure_ascii=False), encoding="utf-8")
    tmp_path.rename(pending_path)
    return {"ok": True, "action_count": len(actions), "pending_path": str(pending_path)}
```

```python
class _CockpitHandler(http.server.BaseHTTPRequestHandler):
    _shell_path: Path = Path(__file__).parent / "cockpit_shell.html"
    _injected: dict = {}  # connector -> list[str html]

    def log_message(self, fmt: str, *args: object) -> None:  # suppress logs
        pass

    def do_OPTIONS(self) -> None:
        self.send_response(204)
        self.send_header("Access-Control-Allow-Origin", "*")
        self.send_header("Access-Control-Allow-Methods", "GET, POST, OPTIONS")
        self.send_header("Access-Control-Allow-Headers", "Content-Type")
        self.end_headers()

    def do_GET(self) -> None:
        path = urllib.parse.urlparse(self.path).path
        if path in ("/", "/index.html"):
            self._serve_shell()
        elif path == "/api/policy":
            self._json(cmd_policy_status())
        elif path == "/api/assets":
            self._json(cmd_assets())
        elif path == "/api/status":
            pending = _resolve_state_root() / "pending-actions.json"
            self._json({"ok": True, "pending": pending.exists()})
        elif path.startswith("/api/components/"):
            self._serve_component(path)
        else:
            self._err(404)

    def do_POST(self) -> None:
        path = urllib.parse.urlparse(self.path).path
        length = int(self.headers.get("Content-Length", 0))
        body: dict = json.loads(self.rfile.read(length)) if length else {}
        if path == "/api/submit":
            self._json(cmd_submit_actions(body.get("actions", [])))
        elif path == "/api/inject-component":
            connector = str(body.get("connector", ""))
            html = str(body.get("html", ""))
            if connector and html:
                _CockpitHandler._injected.setdefault(connector, []).append(html)
            self._json({"ok": True})
        else:
            self._err(404)

    def _serve_shell(self) -> None:
        if not self._shell_path.exists():
            # Fallback minimal page if shell not yet created
            body = b"<html><body><h1>Emerge Cockpit</h1><p>cockpit_shell.html not found</p></body></html>"
            self.send_response(200)
            self.send_header("Content-Type", "text/html")
            self.send_header("Content-Length", len(body))
            self.end_headers()
            self.wfile.write(body)
            return
        body = self._shell_path.read_bytes()
        self.send_response(200)
        self.send_header("Content-Type", "text/html; charset=utf-8")
        self.send_header("Content-Length", len(body))
        self.end_headers()
        self.wfile.write(body)

    def _serve_component(self, path: str) -> None:
        parts = path.strip("/").split("/")  # api / components / connector / filename
        if len(parts) != 4 or ".." in parts[2] or ".." in parts[3]:
            self._err(404)
            return
        connector, filename = parts[2], parts[3]
        if not filename.endswith(".html"):
            self._err(404)
            return
        try:
            fpath = _resolve_connector_root() / connector / "cockpit" / filename
        except Exception:
            self._err(404)
            return
        if not fpath.exists():
            self._err(404)
            return
        body = fpath.read_bytes()
        self.send_response(200)
        self.send_header("Content-Type", "text/html; charset=utf-8")
        self.send_header("Content-Length", len(body))
        self.end_headers()
        self.wfile.write(body)

    def _json(self, data: dict, status: int = 200) -> None:
        body = json.dumps(data, ensure_ascii=False).encode()
        self.send_response(status)
        self.send_header("Content-Type", "application/json")
        self.send_header("Content-Length", len(body))
        self.send_header("Access-Control-Allow-Origin", "*")
        self.end_headers()
        self.wfile.write(body)

    def _err(self, code: int) -> None:
        self.send_response(code)
        self.end_headers()

def cmd_serve(port: int = 0, open_browser: bool = False) -> dict:
    """Start the cockpit HTTP server in a background thread. Returns port and URL."""
    server = socketserver.ThreadingTCPServer(("127.0.0.1", port), _CockpitHandler)
    server.allow_reuse_address = True
    actual_port = server.server_address[1]
    url = f"http://localhost:{actual_port}"
    t = threading.Thread(target=server.serve_forever, daemon=True)
    t.start()
    if open_browser:
        webbrowser.open(url)
    return {"ok": True, "port": actual_port, "url": url}
```

```html
// ── By Status tab ─────────────────────────────────────────────────────────────
function renderStatusTab() {
  if (!policyData) return;
  const pipelines = policyData.pipelines || [];
  const groups = { critical: [], canary: [], stable: [], explore: [] };
  const threshold = policyData.thresholds?.rollback_consecutive_failures || 2;
  pipelines.forEach(p => {
    if (p.consecutive_failures >= threshold || (p.consecutive_failures > 0 && p.success_rate === 0))
      groups.critical.push(p);
    else if (p.status === 'canary') groups.canary.push(p);
    else if (p.status === 'stable') groups.stable.push(p);
    else groups.explore.push(p);
  });

  let html = '';
  const queuedKeys = new Set(queue.map(q => q.data?.key).filter(Boolean));

  if (groups.critical.length) {
    html += `<div class="section-header critical">⚠ Critical (${groups.critical.length})</div>`;
    groups.critical.forEach(p => html += renderPipelineCard(p, queuedKeys));
  }
  if (groups.stable.length) {
    html += `<div class="section-header stable">✓ Stable (${groups.stable.length})</div>`;
    groups.stable.forEach(p => html += renderPipelineCard(p, queuedKeys));
  }
  if (groups.canary.length) {
    html += `<div class="section-header canary">⟳ Canary (${groups.canary.length})</div>`;

    html += `<div class="section-header explore">◯ Explore (${groups.explore.length})</div>`;
    groups.explore.forEach(p => html += renderPipelineCard(p, queuedKeys));
  }
  document.getElementById('main-panel').innerHTML = html;
}

function renderPipelineCard(p, queuedKeys) {
  const key = p.key || '';
  const status = p.status || 'explore';
  const isQueued = queuedKeys.has(key);
  const cf = p.consecutive_failures || 0;
  const sr = p.success_rate != null ? p.success_rate.toFixed(2) : '?';
  const vr = p.verify_rate != null ? p.verify_rate.toFixed(2) : '?';
  const actions = {
    explore: ['— action —', 'promote-canary', 'delete', 'reset-failures'],
    canary:  ['— action —', 'promote-stable', 'demote-explore', 'delete'],
    stable:  ['— action —', 'demote-canary', 'delete'],
  };
  const opts = (actions[status] || actions.explore)
    .map(a => `<option value="${a}">${a === '— action —' ? '— action —' : a}</option>`).join('');
  const cfClass = cf >= 2 ? 'warn' : cf > 0 ? 'warn' : 'good';
  return `
<div class="pipeline-card ${cf >= 2 ? 'critical' : status} ${isQueued ? 'queued' : ''}" data-key="${key}">
  <div class="pipeline-card-row">
    <div class="pipeline-key">${key}</div>
    <span class="pipeline-badge badge-${status}">${status}${p.rollout_pct != null ? ' · ' + p.rollout_pct + '%' : ''}</span>
    <select class="action-select" onchange="onActionSelect(this,'${key}','${status}')">
      ${opts}
    </select>
  </div>
  <div class="pipeline-metrics">
    <span>success: <b class="${parseFloat(sr) >= 0.95 ? 'good' : 'warn'}">${sr}</b></span>
    <span>verify: <b class="${parseFloat(vr) >= 0.95 ? 'good' : 'warn'}">${vr}</b></span>
    <span>failures: <b class="${cfClass}">${cf}</b></span>
    ${p.rollback_executed_count ? `<span>rollbacks: <b class="warn">${p.rollback_executed_count}</b></span>` : ''}
  </div>
</div>`;
}

function onActionSelect(select, key, currentStatus) {
  const action = select.value;
  if (action === '— action —') return;
  select.value = '— action —';

  const actionMap = {
    'promote-canary': { label: 'Promote → canary', type: 'pipeline-set', fields: {status:'canary', rollout_pct:20} },
    'promote-stable': { label: 'Promote → stable', type: 'pipeline-set', fields: {status:'stable', rollout_pct:100} },
    'demote-explore': { label: 'Demote → explore', type: 'pipeline-set', fields: {status:'explore', rollout_pct:0} },
    'demote-canary':  { label: 'Demote → canary',  type: 'pipeline-set', fields: {status:'canary', rollout_pct:20} },
    'reset-failures': { label: 'Reset failures',   type: 'pipeline-set', fields: {consecutive_failures:0} },
    'delete':         { label: 'Delete pipeline',  type: 'pipeline-delete' },
  };
  const def = actionMap[action];
  if (!def) return;

  const item = {
    id: ++queueIdSeq,
    type: def.type,
    label: def.label,
    subLabel: key.replace('pipeline::', ''),
    command: def.type === 'pipeline-delete'
      ? `pipeline-delete ${key}`
      : `pipeline-set ${key} ${JSON.stringify(def.fields)}`,
    data: def.type === 'pipeline-delete'
      ? { type: 'pipeline-delete', key }
      : { type: 'pipeline-set', key, fields: def.fields },
  };
  enqueue(item);
}

function queueNotesComment(connector) {

  if (!comment) return;

function queueScenario(connector, scenarioName, requiredArgs) {
  const args = {};
  for (const arg of requiredArgs) {
    // Only prompt for non-secret-looking args in browser; secrets go via CC terminal
    if (['token', 'password', 'secret', 'key'].some(s => arg.toLowerCase().includes(s))) {
      args[arg] = '(provide via CC)';
    } else {
      const val = prompt(`Arg for scenario "${scenarioName}": ${arg}`);
      if (val === null) return;
      args[arg] = val;
    }
  }
  enqueue({
    id: ++queueIdSeq,
    type: 'scenario-run',
    label: `Run scenario: ${scenarioName}`,
    subLabel: `${connector}`,
    command: `icc_exec intent=write.${connector}.apply-test scenario=${scenarioName}`,
    data: { type: 'scenario-run', connector, scenario: scenarioName, args },
  });
}

// ── Action Queue ──────────────────────────────────────────────────────────────
function enqueue(item) {
  queue.push(item);
  renderQueue();
  if (currentTab === 'status') renderStatusTab();
}

function dequeue(id) {
  queue = queue.filter(q => q.id !== id);
  renderQueue();
  if (currentTab === 'status') renderStatusTab();
}

function clearQueue() {
  queue = [];
  renderQueue();
  if (currentTab === 'status') renderStatusTab();
}

function renderQueue() {
  const count = queue.length;
  document.getElementById('queue-count').textContent = count;
  document.getElementById('submit-btn').disabled = count === 0;
  document.getElementById('submit-btn').textContent = `✓ 提交给 CC 执行 (${count})`;

  const container = document.getElementById('queue-items');
  if (!count) { container.innerHTML = '<div style="color:#484f58;font-size:11px;padding:8px">队列为空</div>'; return; }

  container.innerHTML = queue.map(item => {
    const isDel = item.type === 'pipeline-delete';
    const isPromote = item.label.includes('stable') || item.label.includes('canary');
    const labelClass = isDel ? 'del' : isPromote ? 'promote' : '';
    return `<div class="queue-item ${isDel ? 'delete' : ''}">
      <div class="action-label">
        <b class="${labelClass}">${escHtml(item.label)}</b>
        <span>${escHtml(item.subLabel)}</span>
        <span style="color:#30363d;font-size:9px">${escHtml(item.command.slice(0,50))}</span>
      </div>
      <button class="remove" onclick="dequeue(${item.id})" title="移除">✕</button>
    </div>`;
  }).join('');
}

async function submitQueue() {
  if (!queue.length) return;
  const actions = queue.map(q => q.data);
  document.getElementById('submit-btn').disabled = true;
  document.getElementById('status-msg').textContent = '提交中...';
  try {
    const r = await fetch('/api/submit', {
      method: 'POST',
      headers: {'Content-Type': 'application/json'},
      body: JSON.stringify({ actions }),
    });
    const result = await r.json();
    if (result.ok) {
      queue = [];
      renderQueue();
      document.getElementById('status-msg').textContent = `✓ 已提交 ${result.action_count} 条操作给 CC`;
      if (currentTab === 'status') renderStatusTab();
    } else {
      document.getElementById('status-msg').textContent = '提交失败: ' + JSON.stringify(result);
    }
  } catch(e) {
    document.getElementById('status-msg').textContent = '网络错误: ' + e.message;
```

```python
def test_cockpit_full_flow(tmp_path: Path, monkeypatch):
    """Full flow: start server, submit actions, verify pending-actions.json written."""
    import urllib.request, json

    # Set up connector with NOTES and scenario
    connector_root = tmp_path / "connectors"
    (connector_root / "myconn" / "scenarios").mkdir(parents=True)
    (connector_root / "myconn" / "NOTES.md").write_text("# Notes\nsome info", encoding="utf-8")
    (connector_root / "myconn" / "scenarios" / "test.yaml").write_text(
        "name: test\ndescription: a test\nsteps:\n  - name: s1\n    type: http_get\n    base_url: '{{ env_url }}'\n    path: /health\n",
        encoding="utf-8",
    )
    monkeypatch.setenv("EMERGE_CONNECTOR_ROOT", str(connector_root))
    monkeypatch.setenv("EMERGE_REPL_ROOT", str(tmp_path))

    from scripts.repl_admin import cmd_serve
    r = cmd_serve(port=0, open_browser=False)
    base = r["url"]

    # Assets endpoint includes connector
    with urllib.request.urlopen(f"{base}/api/assets") as resp:
        assets = json.loads(resp.read())
    assert "myconn" in assets["connectors"]
    assert assets["connectors"]["myconn"]["notes"] is not None
    assert len(assets["connectors"]["myconn"]["scenarios"]) == 1

    # Submit actions
    actions = [
        {"type": "notes-comment", "connector": "myconn", "comment": "test comment"},
        {"type": "scenario-run", "connector": "myconn", "scenario": "test", "args": {"env_url": "http://localhost"}},
    ]
    body = json.dumps({"actions": actions}).encode()
    req = urllib.request.Request(
        f"{base}/api/submit", data=body,
        headers={"Content-Type": "application/json"}, method="POST"
    )
    with urllib.request.urlopen(req) as resp:
        result = json.loads(resp.read())
    assert result["ok"]

    # pending-actions.json should be written
    pending = json.loads((tmp_path / "pending-actions.json").read_text())
    assert len(pending["actions"]) == 2
    assert pending["actions"][0]["type"] == "notes-comment"
    assert pending["actions"][1]["type"] == "scenario-run"
```
